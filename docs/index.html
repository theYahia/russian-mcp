Быстрый старт

Добавь в конфиг Claude Desktop, Cursor или VS Code

  "mcpServers": {
    "dadata": {
      "command": "npx",
      "args": ["-y", "@metarebalance/dadata-mcp"],
      "env": {
        "DADATA_API_KEY": "your-key"
      }
    }
  }
}
