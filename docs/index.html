Quick Start

Add to your Claude Desktop, Cursor, or VS Code config

  "mcpServers": {
    "dadata": {
      "command": "npx",
      "args": ["-y", "@metarebalance/dadata-mcp"],
      "env": {
        "DADATA_API_KEY": "your-key"
      }
    }
  }
}
